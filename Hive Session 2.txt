#Dynamic Partition 
enable this partition 


create table cust_np (cid int,cname string,city string)
row format delimited
fields terminated by ',';

load data local inpath '/home/hadoop/cust.txt' into table cust_np;

create table cust_dp (cid int,cname string)
partitioned by (city string)
row format delimited
fields terminated by ',';

set hive.exec.dynamic.partition.mode=nonstrict;

insert into cust_dp partition(city) select * from cust_np;
