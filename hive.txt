#partition
Problem specification:If u have partition data by city,while getting data only for pune will check only pune partition...hence performance improved
is a performace optimization technique,storing data in partitions.
whenever following partitiing commands os performed data will get stored in parts.
Lets see practically...
COMMAND:
# internal table
create table cust (cid int,cname string,city string)
row format delimited
fields terminated by '|';
#for partition
create table cust (cid int,cname string)
partition by (city string)
row format delimited
fields terminated by '|';
data of city will be saved in partition with different location by city
#for loading data in partition table
2 methods(static and dynamic)
`#static partition(enabled default):used to bulk load(i.e.large data)
 here we need to load 1 by 1 partition manually
 FOR STATIC PARTITION U NEED SEPARATE FILES FOR SEPARATE PARTITION (e.g.Pune.txt,Mumbai.txt,Surat.txt...)
COMMANDS:
load data local inpath '/home/hadoop/pune.txt' into cust_p  partition(city='Pune');
load data local inpath '/home/hadoop/mumbai.txt' into cust_p  partition(city='Mumbai');
load data local inpath '/home/hadoop/surat.txt' into cust_p  partition(city='Surat');
    if files are on HDFS use only (..data inpath '...)
    after execution Pune folder will created and record will get stored inside it.
    user/hive/warehouse/cust_p/city=Pune/pune.txt
    user/hive/warehouse/cust_p/city=Mumbai/mumbai.txt
    user/hive/warehouse/cust_p/city=Surat/surat.txt
                             
show partition cust_p;
output:
    OK
    city=Pune
    city=Mumbai
    city=Surat

`#dynamic partition
FOR FOR DYNAMIC PARTITION U NEED  FILE FOR  PARTITION (e.g.cust_p.txt[which includes data of all cities])
Steps:1)Create non-partition table load data inside it
2)insert data from non-partitoin table to partition table

COMMANDS:

Lets create partition table:
create table cust_dp (cid int,cname string)

#Bucketing
bucketing is similar to partition but difference is when column have too many variety of data,for e.g.Mobile no.
partitioning mobile number will create separate folder for each record,instead we can use bucketing
bucketing creates files.(bucketing create files,partiton create folders)
when to use:
    1.not particualr column for partition
    2.unequal partition size
Bucketing works on hashing technique,u just have to determine number of buckets
buckets are determined using modulous(%)i.e. reminder after division
https://www.javatpoint.com/bucketing-in-hive
e.g.
cust 
cid cname

2%4 = 2

----

5%4   1
6%4   2
-------
7%4   3
8%4   0
-----------
9%4   1
10%4  2


1 5 9
-----
2 6 10
----------
3 7
----------
4 8


COMMAND:
create table cust_b(cid int,cname string)
clustered by (cid) into 4 buckets
row format delimited
fields terminated by '|';

insert into cust_b select * from cust;
(given cust have all customer data seprated by pipe'|')

Note:bucketing is not limited to numerical column,even string can be bucketed using hashing technique

#Complex datatype handling in hive
premitive datatypes:
non-primitive datatype:
    Array|Map|struct
##Array:
    collection of multiple element of same datatype
    e.g.table stud:
    id|name|marks
    1|rahul|24,52,64,23
COMMAND:
    create table stud(id int,name string,marks array<int>)
    row format delimited
    fields terminated by '|'
    collection items terminated by ',';

    <load data>(refer above for this command)

   to access array
    select id,name,marks[0] as marathi from stud;
    NOTE:[-1] is not supported
    output:
    id name marathi
    1 rahul 24

##struct:
    collection of multiple element of different datatype
    e.g.table 
    id|name|address
    1|rahul|CST,Mumbai,424242
COMMAND:
    create table stdnt(id int,name string,address struct<street:string,city:string,pincode:int>)
    row format delimited
    fields terminated by '|'
    collection items terminated by ',';

    <load data >

    to access struct
    select id,name,address.street,address.city,address.pincode from stdnt;
    output:
    id name street city pincode
    1 rahul CST Mumbai 424242

##map:
    collection of data in key-value pair
    e.g.table info
    cid,cname,address
    1,rahul,padd:pune|cadd:mumbai
    2,rohan,padd:surat
    3,rock,padd:delhi|cadd:surat
COMMAND:
    create table info(cid int,cname string,address map<string,string>)
    row format delimited
    fields terminated by ','
    collection items terminated by '|'
    map keys terminated by ':';

    <load data>

    to access map
    select cid,cname,address['padd'] as permaddress from info;
    output:
    1 rahul pune
    2 rohan surat
    3 rock delhi

#7)MAP JOIN / BROADCAST JOIN
all normal join operation will work in hive i.e.inner,left,right,full,but takes more time ,hence this concept.
Use this JOIN with condition any one table should be smaller,cause the data will temporary stored in RAM/MEMORY.
//creating table
create table lcn (city string , area string)
row format delimited
fields terminated by ',';
//data
lcn.txt

city,area
pune,ABC
mumbai,CST
delhi,JM
//loading data
load data local inpath '/home/hadoop/lcn.txt' into table lcn;
//NORMAL join
select * from cust inner join lcn on cust.city=lcn.city;
//MAPJOIN
select /*+ MAPJOIN(lcn) */ cid,cname,cust.city,area from cust inner join lcn on cust.city=lcn.city;
 here,the column inside MAPJOIN(_ _ _) is loaded inside memory/ram/shared cache
    /*+ MAPJOIN(...) */ is called map hint/broadcast hint
    when next time join used it will fetch from memory.
the size of smaller table default is 10mb,here table lcn should be below 10mb
    size is customizable...max recomended 256mb.
if u r not aware of which table is smaller,then another concept is AUTO MAP JOIN
//AUTO MAP JOIN
    setting property
set hive.auto.convert.join=true;
    no need for map hint/broadcast hint here
select * from cust inner join lcn on cust.city=lcn.city;
    it will automatic select smaller table.

#VIEWS IN HIVE
    temporary table are views just like in sql
purpose:view are create on top of table to abstract away data and give only particular access
(security,hiding complexity)
we can save complex select statement queries as views.
SYNTAX:
    create view <view_name> as select ... from <tablename>
2 types
1.simple view:
 includes simple select statements(where clause,)
2.complex view:
 includes complex select statements(like group by,join,windowing functions,etc.)

#9)Vectorization(another performance improvement technique):
    Vertical scanning /horizontal scanning
    normally hive reads data row by row 
    but if u enable Vectorization=true property
    performance will increase because it will scan multiple columns at a time
    default is 1024 records at once(customizable)
COMMAND:
    set hive.enable.vectorization=true;
