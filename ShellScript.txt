#parameter passing at runtime

myscript.sh
	#!/bin/bash
  dir_name=$1
  mkdir ~/dir_name
  echo $1 folder is created.

bash myscript.sh TEST      #"TEST" is parameter passed to $1 in script

output:TEST folder is created

similarly,u can pass $1,$2,$3,$4,... in script as 
bash script.sh var1 var2 var3 ...
$0 stored the filename by default

#if...else(conditional statements)
> -gt
>= -ge
< -lt
<= -le
== -eq

script.sh
	#!/bin/bash
	name=karan
	age=10
	if [ $age -gt 18 ]
	then
	echo "you are eligible for voting"
	else
	echo "you are not eligible for voting"
	fi
	echo "script completed"

-f #option-checks the file inside if statement in script

#!/bin/bash
if [ -f "/home/ubuntu/data.txt" ]
then
echo "file exist"
else
echo "file not exist"
fi 
echo "successful"
	
-d #option-checks the dir inside if statement in script

#!/bin/bash
if [ -d "/home/ubuntu/test1" ]
then
echo "dir exist"
else
mkdir /home/ubuntu/test1
fi 
echo "successful"
